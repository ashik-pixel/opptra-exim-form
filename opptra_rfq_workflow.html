Setup Guide — Connect to Google Sheets

This takes about 5 minutes. Every form submission will add a new row to your Google Sheet and send an email notification to your EXIM team.
1
Create a Google Sheet

Go to sheets.google.com → create a new blank sheet. Name it "Opptra EXIM Submissions".

2
Open Apps Script

In your Sheet, click Extensions → Apps Script. A new tab opens with a code editor. Delete all existing code.

3
Paste this script

Copy the full script below and paste it into the editor:

const SHEET_NAME = "Sheet1";
const NOTIFY_EMAIL = "your-email@opptra.com"; // Change this

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME);

    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
    }

    // Build header row on first submission
    if (sheet.getLastRow() === 0) {
      const headers = Object.keys(data);
      sheet.appendRow(headers);
      // Bold + freeze header
      sheet.getRange(1, 1, 1, headers.length)
        .setFontWeight("bold")
        .setBackground("#131A48")
        .setFontColor("#FFFFFF");
      sheet.setFrozenRows(1);
    }

    // Append data row
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const row = headers.map(h => data[h] || "");
    sheet.appendRow(row);

    // Auto-resize columns
    for (let i = 1; i <= headers.length; i++) {
      sheet.autoResizeColumn(i);
    }

    // Email notification
    if (NOTIFY_EMAIL && NOTIFY_EMAIL !== "your-email@opptra.com") {
      const subject = "New EXIM Request: " + (data.ref_number || "N/A");
      let body = "New shipment request from Opptra EXIM form.\n\n";
      for (const [k, v] of Object.entries(data)) {
        body += k.replace(/_/g, " ") + ": " + v + "\n";
      }
      MailApp.sendEmail(NOTIFY_EMAIL, subject, body);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ status: "ok", row: sheet.getLastRow() }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ status: "ready", message: "Opptra EXIM endpoint active" }))
    .setMimeType(ContentService.MimeType.JSON);
}
Change your-email@opptra.com on line 2 to the actual EXIM team email before deploying.
4
Deploy as a Web App

Click Deploy → New Deployment. Set type to Web app. Set "Execute as" to Me. Set "Who has access" to Anyone. Click Deploy. Authorize when prompted.

5
Copy the Web App URL & paste it in the setup banner

Google will show a URL like https://script.google.com/macros/s/.../exec. Copy it and paste it into the setup banner at the top of this page. Click Save & Test.

Done! Every submission will now save a row in your Google Sheet. If you set the email, you'll also get a notification per submission.
